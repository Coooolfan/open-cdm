#define "allow" as string
        default "true"
        enum ["true", "false"]
        hint "是否允许使用 WITH 字句"

if cast(#{allow} as bool) then
  return true
end

if @domain.sqlType == 'DELETE' then
  return not @domain.hasWith
end

if @domain.sqlType == 'SELECT' and @domain.mode == 'INSERT' then
  return not @domain.hasWith
end
