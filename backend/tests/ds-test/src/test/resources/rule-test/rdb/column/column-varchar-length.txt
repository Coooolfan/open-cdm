#define "allowEmpty" as bool
        default "false"
        enum ["true", "false"]
        hint "是否允许空长度"
#define "length" as int
        default "120"
        hint "字段最大长度"

if
  @domain.sqlType not in ['CREATE_TABLE_ADD_COLUMN', 'ALTER_TABLE_ADD_COLUMN', 'ALTER_TABLE_ALTER_COLUMN']
then
  return true
end

if
  @func.string.startsWith(@domain.typeDesc, "varchar") or
  @func.string.startsWith(@domain.typeDesc, "nvarchar")
then

  if
    @func.string.isBlank(@domain.length) and
    cast(#{allowEmpty} as bool)
  then
    return true
  end

  if @func.string.isBlank(@domain.length) then
    return false // 语句中长度为空，例如 create table abc(name varchar);
  else
    return cast(@domain.length as int) <= cast(#{length} as int)
  end

end
