#define "allow" as string
        default "true"
        enum ["true", "false"]
        hint "表示是否允许联表查询"

if
  @domain.sqlType in ['SELECT']
then
  if #{allow} == 'true'
  then
     return true
  end

  if @func.string.equalsAny( @domain.dsType , ['Db2','Db2Fori','SQLServer']) == false
    then
      if @func.array.size(@domain.joinTypes) > 0
      then
          return false
      else
          return true
      end
    else
      if
         @domain.joinType == 'NONE'
      then
        return true
      else
        return false
      end
    end

end
