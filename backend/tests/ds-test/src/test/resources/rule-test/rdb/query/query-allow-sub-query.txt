#define "allow" as string
        default "Any"
        enum ["Any", "Select", "From", "Where", "None"]
        hint "允许子查询出现的位置"

if
  @domain.sqlType != 'SELECT' or
  @domain.mode != 'NORMAL'
then
  return true
end

if 'Any' in @func.string.split(#{allow}, ',') then
  return true
end

if
  @domain.selectInSelect and
  'Select' not in @func.string.split(#{allow}, ',')
then
  return false
end

if
  @domain.selectInFrom and
  'From' not in @func.string.split(#{allow}, ',')
then
  return false
end

if
  @domain.selectInWhere and
  'Where' not in @func.string.split(#{allow}, ',')
then
  return false
end
