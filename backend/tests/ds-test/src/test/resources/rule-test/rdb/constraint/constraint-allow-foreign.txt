#define "allow" as bool
        default "false"
        enum ["true", "false"]
        hint "是否允许使用外键约束"

if
  @domain.sqlType in ['CREATE_TABLE_ADD_CONSTRAINT', 'ALTER_TABLE_ADD_CONSTRAINT']
then

  if @domain.type == 'ForeignKey' then
    return cast(#{allow} as bool)
  else
    return true
  end

end
