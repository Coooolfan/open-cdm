  SELECT user_id, COUNT(*) AS cnt
  FROM posts
  WHERE published = TRUE
  GROUP BY user_id
)
SELECT u.id, u.username, rp.cnt,
  ROW_NUMBER() OVER (ORDER BY rp.cnt DESC) AS rank1
FROM users u
LEFT JOIN recent_posts rp ON rp.user_id = u.id
ORDER BY rp.cnt DESC
LIMIT 10;
